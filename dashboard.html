async function loadClubDetails(shortName) { const badgePath = `images/club_badges/${shortName}.png`; // Fetch club by ShortName const { data: clubData } = await window.supabaseClient .from("Clubs") .select("Club") .eq("ShortName", shortName) .single(); const fullName = clubData.Club; const clubId = fullName; // Club name is now the PK and FK // Fetch players contracted to this club const { data: players } = await window.supabaseClient .from("Players") .select("*") .eq("Contracted_Team", fullName); // Fetch listings for this club const { data: listingsData } = await window.supabaseClient .from("Player_Transfer_Listings") .select("*") .eq("seller_club_id", fullName); const listings = listingsData || []; const listedSet = new Set(listings.map(l => String(l.player_id))); const positionOrder = { "GK": 1, "LB": 2, "CB": 3, "RB": 4, "DMF": 5, "LMF": 6, "CMF": 7, "RMF": 8, "AMF": 9, "LWF": 10, "CF": 11, "SS": 12, "RWF": 13 }; players.sort((a, b) => (positionOrder[a.Position] || 999) - (positionOrder[b.Position] || 999) ); document.getElementById("dashboardTitle").innerHTML = ` ${fullName} Dashboard `; let html = `

${fullName}

Squad

`; players.forEach(p => { const isListed = listedSet.has(String(p.Konami_ID)); html += ` `; }); html += `
Name Pos Nation Age Rating Style Value List Player
${p.Name} ${p.Position} ${p.Nation} ${p.Age} ${p.Rating} ${p.Playstyle} ${formatBTC(p.market_value)} ${ isListed ? `🔒 LISTED` : `` }

Your Listed Players

`; listings.forEach(l => { const player = players.find(p => String(p.Konami_ID) === String(l.player_id)); if (!player) return; const timeLeft = timeRemaining(l.end_time); html += ` `; }); html += `
Name Pos Rating Reserve Time Remaining Status
${player.Name} ${player.Position} ${player.Rating} ${formatBTC(l.reserve_price)} ${timeLeft} Active
`; document.getElementById("clubArea").innerHTML = html; document.querySelectorAll(".listPlayer").forEach(cb => { cb.addEventListener("change", async () => { if (!cb.checked) return; const name = cb.dataset.name; const id = cb.dataset.id; const mv = Number(cb.dataset.mv); const reserve = Math.round(mv * 1.5); const confirmMsg = ` You are about to list ${name} for transfer. This action is permanent. There are NO take-backs. The reserve price will be automatically set to: ${formatBTC(reserve)} If any bid meets or exceeds this amount within 24 hours, the player WILL be sold automatically. Are you absolutely sure? `; if (!confirm(confirmMsg)) { cb.checked = false; return; } await window.supabaseClient .from("Player_Transfer_Listings") .insert({ player_id: String(id), seller_club_id: fullName, reserve_price: reserve, listing_type: "standard", hidden_bids: false, special_rules: {}, start_time: new Date().toISOString(), end_time: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(), current_highest_bid: 0, current_highest_bidder: null }); alert("Player listed successfully."); loadClubDetails(shortName); }); }); }
